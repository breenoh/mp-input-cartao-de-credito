para pegar os dados em tempo real, usar eventlistener keyup do teclado para ir trocando no cartao tambem

let cardNumber = document.querySelector("#card-number")
let numDigitado
numeroCartao.addEventListener('input', event => {
    numDigitado = event.target.value
    for(let i = 0; numDigitado.lenght; i++){
        cardNumber = cardNumber.replace('x', numDigitado[i])
    }
})
